// Save as setup.js and run: node setup.js const { execSync } = require('child_process'); const fs = require('fs'); const path = require('path'); const config = { OWNER_NUMBER: '234XXXXXXXXXX', // CHANGE THIS BOT_NAME: 'TUNZY-MD', OWNER_NAME: 'YourName' }; try { console.log('Cloning TUNZY-MD...'); execSync('git clone https://github.com/tunzy-shop/TUNZY-MD temp-dir', { stdio: 'inherit' }); // Move files const files = fs.readdirSync('temp-dir'); for (const file of files) { fs.renameSync(`temp-dir/${file}`, file); } fs.rmdirSync('temp-dir'); // Update owner.json const ownerPath = 'data/owner.json'; if (fs.existsSync(ownerPath)) { let ownerData = JSON.parse(fs.readFileSync(ownerPath)); if (!Array.isArray(ownerData)) ownerData = []; ownerData.push(`${config.OWNER_NUMBER}@s.whatsapp.net`); fs.writeFileSync(ownerPath, JSON.stringify(ownerData, null, 2)); } console.log('Installing dependencies...'); execSync('npm install', { stdio: 'inherit' }); console.log('Starting bot...'); execSync('npm start', { stdio: 'inherit' }); } catch (err) { console.error('Setup failed:', err.message); }
